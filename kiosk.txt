@echo off
setlocal enabledelayedexpansion

:: Configuration Paths
set "ONLINE_URL=https://language-kiosk-mode.vercel.app/"
set "OFFLINE_FILE=file:///C:/kiosk/offline.html"
set "STATUS=online"

:: Move inside the official Chrome directory to prevent index page bugs
cd /d "C:\Program Files\Google\Chrome\Application"

:loop
:: Check network status by pinging Google DNS
ping -n 1 8.8.8.8 >nul 2>&1

if errorlevel 1 (
    :: Internet is DOWN
    if "!STATUS!"=="online" (
        taskkill /F /IM chrome.exe >nul 2>&1
        timeout /t 1 /nobreak >nul
        start "" "chrome.exe" --kiosk --incognito --disable-popup-blocking --disable-desktop-notifications --deny-permission-prompts --disk-cache-size=1 --media-cache-size=1 "%OFFLINE_FILE%"
        set "STATUS=offline"
    )
) else (
    :: Internet is UP
    if "!STATUS!"=="offline" (
        taskkill /F /IM chrome.exe >nul 2>&1
        timeout /t 1 /nobreak >nul
        start "" "chrome.exe" --kiosk --incognito --disable-popup-blocking --disable-desktop-notifications --deny-permission-prompts --disk-cache-size=1 --media-cache-size=1 "%ONLINE_URL%"
        set "STATUS=online"
    )
)

:: Safety double-check to make sure Chrome is always running something
tasklist /FI "IMAGENAME eq chrome.exe" | findstr /I "chrome.exe" >nul
if errorlevel 1 (
    if "!STATUS!"=="online" (
        start "" "chrome.exe" --kiosk --incognito --disable-popup-blocking --disable-desktop-notifications --deny-permission-prompts --disk-cache-size=1 --media-cache-size=1 "%ONLINE_URL%"
    ) else (
        start "" "chrome.exe" --kiosk --incognito --disable-popup-blocking --disable-desktop-notifications --deny-permission-prompts --disk-cache-size=1 --media-cache-size=1 "%OFFLINE_FILE%"
    )
)

timeout /t 3 /nobreak >nul
goto loop
