    <script>
        // Set your desired PIN here
        const CORRECT_PIN = "4445"; 
        let currentInput = "";

        function togglePinPad(show) {
            document.getElementById('pinOverlay').style.display = show ? 'flex' : 'none';
            clearPin();
        }

        function pressKey(num) {
            if (currentInput.length < 4) {
                currentInput += num;
                updateDisplay();
                
                if (currentInput === CORRECT_PIN) {
                    window.close();
                } else if (currentInput.length === 4) {
                    // Flash red or clear if wrong
                    setTimeout(clearPin, 500);
                }
            }
        }

        function clearPin() {
            currentInput = "";
            updateDisplay();
        }

        function updateDisplay() {
            let dots = "";
            for(let i=0; i < currentInput.length; i++) {
                dots += "●";
            }
            document.getElementById('pinDisplay').innerText = dots || "****";
        }
    </script>
